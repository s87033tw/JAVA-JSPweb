import java.util.*;
import java.io.*;


class load{


	public static void main(String[]args)throws Exception{

		ArrayList<String>status=new ArrayList<>();	
		ArrayList<String>computernum=new ArrayList<>();
		File f1=new File("save.txt");
		FileReader fr=new FileReader(f1);
		BufferedReader br=new BufferedReader(fr);
		String statusmessage="";

		int statuscount=0;
		String message="";
		//String message2="";
		int length=0;
		while((statusmessage=br.readLine())!=null){

			length=statusmessage.indexOf(" ");
			for(int i=0;i<length;i++){

				message+=statusmessage.charAt(i)+"";				

			}
			computernum.add(message);
			message="";
			for(int j=length+1;j<statusmessage.length();j++){

				message+=statusmessage.charAt(j)+"";

			}
			status.add(message);
			message="";
		}
		for(int i=0;i<status.size();i++){
			
			System.out.println(computernum.get(i)+"\t"+status.get(i));


		}

			

	}


}
